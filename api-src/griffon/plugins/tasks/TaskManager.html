16 package griffon.plugins.tasks;
17 
18 import java.util.concurrent.Callable;
19 import java.util.concurrent.Future;
20 
21 
22 /**
23  * @author <a href="mailto:eike.kettner@gmail.com">Eike Kettner</a>
24  * @since 19.07.11 22:16
25  */
26 public interface TaskManager {
27 
28     /**
29      * Creates a new {@link TaskControl} for a given task. The state is
30      * set to {@link griffon.plugins.tasks.Task.State#PENDING} and the task is not started. That
31      * allows to add listeners for this task execution before starting
32      * it.
33      * <p/>
34      * Tasks created by this method will fire {@link TaskEvent}s as
35      * appropriate. See {@link TaskListener} for details.
36      *
37      * @param task
38      * @param <V>
39      * @param <C>
40      * @return
41      */
42 
43     <V, C> TaskControl<V> create(Task<V, C> task);
44 
45     /**
46      * Allows to add listeners.
47      *
48      * @return
49      */
50 
51     TaskListenerSupport getTaskListenerSupport();
52 
53     /**
54      * Returns a collection of all currently created task conforming to
55      * the specified predicate. The contract is to have all started
56      * and pending tasks available. It's not necessary returning finished
57      * tasks.
58      *
59      * @param predicate
60      * @return
61      */
62 
63     Iterable<TaskControl> getTasks(TaskPredicate predicate);
64 
65 
66     TaskControl findTask(TaskPredicate predicate);
67 
68     /**
69      * Looks up a task by the specified context id.
70      *
71      * @param contextId
72      * @return
73      */
74 
75     TaskControl getTask(String contextId);
76 
77     /**
78      * Submits a {@link Runnable} object for execution. Contrary to executing
79      * {@link Task}s this is more lightweight as it does not fire any events
80      * and is not managed by this manager.
81      *
82      * @param task
83      * @return
84      */
85 
86     Future<?> submit(Runnable task);
87 
88     /**
89      * Submits a {@link java.util.concurrent.Callable} object for execution. Contrary to executing
90      * {@link Task}s this is more lightweight as it does not fire any events
91      * and is not managed by this manager.
92      *
93      * @param task
94      * @return
95      */
96 
97     <T> Future<T> submit(Callable<T> task);
98 
99 }
